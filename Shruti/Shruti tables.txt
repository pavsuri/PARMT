CREATE TABLE `activities` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `title` varchar(50) NOT NULL,
  `description` varchar(500) NOT NULL,
  `activity_date` int(11) NOT NULL,
  `image_name` varchar(100) NOT NULL,
  `image_type` varchar(3) NOT NULL,
  `image_path` varchar(100) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=latin1



CREATE TABLE `categories` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `name` varchar(100) NOT NULL,
  `mark_deleted` int(11) NOT NULL DEFAULT '0',
  `created_on` date NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=latin1



CREATE TABLE `adminuser` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `name` int(11) NOT NULL,
  `username` varchar(100) NOT NULL,
  `password` varchar(250) NOT NULL,
   `gmail` varchar(250) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=latin1


CREATE TABLE `user_enquiries` (
  `id` int(5) NOT NULL AUTO_INCREMENT,
  `user_id` int(5) NOT NULL,
  `description` varchar(500) NOT NULL,
  `enq_date` date NOT NULL,
  `admin_user_id` int(3) NOT NULL,
  `reply_message` varchar(500) NOT NULL,
  `reply_date` date NOT NULL,
  `mark_completed` tinyint(4) NOT NULL,
  `mark_deleted` tinyint(4) NOT NULL DEFAULT '0',
  PRIMARY KEY (`id`)
)
